ADD TRIGGERS TO GET WARNING

DELIMITER $$ 
CREATE TRIGGER languages_number_test
BEFORE Insert
ON countrylanguage FOR EACH ROW
BEGIN
    DECLARE total int;
Select count(language) into total from countrylanguage where new.countrycode = countrycode;
                        
    IF total>=9 THEN
        SIGNAL SQLSTATE '45000' 
            SET MESSAGE_TEXT = `Warning: languages enough for this country “-” `;
    END IF;
END $$ 
DELIMITER ;
